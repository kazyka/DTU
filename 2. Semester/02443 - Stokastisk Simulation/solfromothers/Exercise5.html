def stats(array,alpha = 0.05):
    n = len(array)
    mean = np.mean(array)
    #var = 1/(n-1)*(sum([i**2 for i in array])-n*mean**2)
    sd = np.var(array)**(1/2)
    lower,upper =mean + sd*scipy.stats.t.ppf([alpha/2,1-alpha/2],n-1)
    stat = pd.DataFrame(np.array([[mean,sd,upper, lower]]))
    stat.columns = ['Mean','Standard diviation','Upper','Lower']
    display(stat)
    plt.plot(array,label='Monte Carlo Estimator $\hat{\\theta}$')
    plt.plot([0,100],[mean,mean],'--',c='gray',label='Mean')
    plt.plot([0,100],[upper,upper],'--r',label='95\% CI')
    plt.plot([0,100],[lower,lower],'--r')
    plt.legend()
    plt.ylabel("Monte Carlo Estimate")
    plt.xlabel("Run")
    plt.show()

Stratified sampling¶
This is a general survey technique: We sample in predetermined areas, using knowledge of structure of the sampling space
\begin{equation}
W_i = \frac{e^\frac{U_{i,1}}{10}+e^{\frac{1}{10}+\frac{U_{i,2}}{10}}+e^{\frac{2}{10}+\frac{U_{i,3}}{10}}+...+e^{\frac{9}{10}+\frac{U_{i,10}}{10}}}{10}
\end{equation}
